                downloads: 0,
                citations: 0
            };

            papers.unshift(newPaper);
            displayPapers(papers);
            updateStats();
            closePublishModal();
            
            // Show success message
            alert('Your research has been published successfully!');
        }

        // Login function
        function loginUser(event) {
            event.preventDefault();
            alert('Demo: Login functionality would be implemented with backend authentication.');
            closeLoginModal();
        }

        function switchToSignup() {
            closeLoginModal();
            alert('Demo: Sign up functionality would redirect to registration page.');
        }

        // Search papers
        function searchPapers() {
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            
            if (!searchTerm) {
                displayPapers(papers);
                return;
            }

            const filtered = papers.filter(paper => 
                paper.title.toLowerCase().includes(searchTerm) ||
                paper.authors.some(author => author.toLowerCase().includes(searchTerm)) ||
                paper.abstract.toLowerCase().includes(searchTerm) ||
                paper.tags.some(tag => tag.toLowerCase().includes(searchTerm))
            );

            displayPapers(filtered);
        }

        // Filter by category
        function filterByCategory(category) {
            const filtered = papers.filter(paper => paper.category === category);
            displayPapers(filtered);
            
            // Highlight selected category
            document.querySelectorAll('.category-card').forEach(card => {
                card.style.borderColor = card.querySelector('.category-name').textContent === category ? '#2563eb' : 'transparent';
            });
        }

        // Download paper
        function downloadPaper(paperId) {
            const paper = papers.find(p => p.id === paperId);
            if (paper) {
                paper.downloads++;
                alert(`Downloading: ${paper.title}\nIn a production environment, this would download the PDF file.`);
                updateStats();
            }
        }

        // Cite paper
        function citePaper(paperId) {
            const paper = papers.find(p => p.id === paperId);
            if (paper) {
                paper.citations++;
                const citation = `${paper.authors.join(', ')} (${paper.date}). ${paper.title}. ${paper.institution}.`;
                navigator.clipboard.writeText(citation).then(() => {
                    alert('Citation copied to clipboard!');
                });
                updateStats();
            }
        }

        // Close modals when clicking outside

                event.target.classList.remove('active');
            }
        };
